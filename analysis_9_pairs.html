def get_key_ideas(df):
    df['Cleaned joined methods'] = df['Cleaned method 1'].map(str) + ' | ' + df['Cleaned method 2']
    submitted_ideas = set(df[df['Selection'].str.contains('submit',na=False)]['Cleaned joined methods'])
    deleted_all_ideas = set(df[df['Selection'].str.contains('deleteAll',na=False)]['Cleaned joined methods'])
    ideas = deleted_all_ideas | submitted_ideas
    ideas = sorted([str(x) for x in list(ideas)])
    ideas_in_time = []
    for i in ideas:
        ideas_in_time.append((min(df[df['Cleaned joined methods'].str.contains(i,regex=False)]['Time_seconds']),i))
    ideas_in_time = sorted(ideas_in_time, key = lambda element : element[0])
    return ideas_in_time
